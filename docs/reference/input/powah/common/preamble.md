## Usage

```lua-vue{4}
local telem = require 'telem'

local backplane = telem.backplane()
  :addInput('my_{{ $frontmatter?.telem?.adapter?.id || 'ADAPTER.ID' }}', telem.input.powah.{{ $frontmatter?.telem?.adapter?.id || 'ADAPTER.ID' }}('right', '*'))
  :cycleEvery(5)()
```
